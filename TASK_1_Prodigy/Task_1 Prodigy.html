#Select a column name to plot here we have choosen '1998'
choosen_year = '1998'
# x axis is taken as population & y axis is Country name.
data_for_year = df[['Country Name', choosen_year]]
#Ascending (denotes the order preference can be True/False)
data_for_year = data_for_year.sort_values(by = choosen_year, 
                                          ascending = True)
#Plotting the top 25 alone 
data_for_year_top25 = data_for_year.head(25)
plt.figure(figsize = (12,10))
sns.barplot(y = 'Country Name', 
            x = choosen_year, 
           data = data_for_year_top25, 
            orient = 'h')
plt.title('TOP 25(1998)')
plt.xlabel('Population')
plt.ylabel('Country Name')
plt.show()
